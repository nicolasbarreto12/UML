package Logica;

public class Avion {
String modelo;
String Patente;
int capacidad;

public Avion(String modelo, String patente, int capacidad) {
	this.modelo = modelo;
	Patente = patente;
	this.capacidad = capacidad;
}
public String getModelo() {
	return modelo;
}
public void setModelo(String modelo) {
	this.modelo = modelo;
}
public String getPatente() {
	return Patente;
}
public void setPatente(String patente) {
	Patente = patente;
}
public int getCapacidad() {
	return capacidad;
}
public void setCapacidad(int capacidad) {
	this.capacidad = capacidad;
}

}
